Sxy = np.sum(x*y)
Sx  = np.sum(x)
Sy  = np.sum(y)
Sxx = np.sum(x*x)
Fac = n*Sxx-Sx*Sx
m   = (n*Sxy-Sx*Sy)/Fac
b   = (Sxx*Sy-Sx*Sxy)/Fac
print ('m = {0:.2E} cm/g'.format(m))
print ('b = {0:.2E} cm'.format(b))
m = 1.17E-03 cm/g
b = 1.27E+01 cm
