Datos = array(
[[20, 0.20104],

\[\begin{align} m &= \frac{\displaystyle n\sum_{i=1}^n x_i y_i -\sum_{i=1}^{n} x_i\sum_{i=1}^n y_i}{\displaystyle n\sum_{i=1}^n x_i^2-\bigg( \sum_{i=1}^n x_i \bigg)^2}\\ m &= 0.009749142857 \end{align}\]

Code 
X = Datos[:,0]
Y = Datos[:,1]
n = len(X)

SX  = sum(X)
SY  = sum(Y)
SXY = sum( X*Y )
SXX = sum(X*X)

m = (n*SXY-SX*SY)/( n*SXX-SX*SX )
print("m=",m)
m= 0.009749142857142859

\[\begin{align} b &= \frac{\displaystyle \sum_{i=1}^{n} x_i^2 \sum_{i=1}^{n} y_i - \sum_{i=1}^{n} x_i\sum_{i=1}^{n} x_i y_i}{\displaystyle n\sum_{i=1}^n x_i^2-\bigg( \sum_{i=1}^n x_i \bigg)^2}\\ b &= 0.00892 \end{align}\]

Code 
b = (SXX*SY - SX*SXY)/( n*SXX-SX*SX )
print("b=",b)
b= 0.008919999999999739

\[\begin{align} u(m) &= \sqrt{ \frac{\displaystyle n\sum_{i=1}^n (y_i-mx_i-b)^2}{\displaystyle (n-2) \bigg[ n\sum_{i=1}^n x_i^2-\bigg( \sum_{i=1}^n x_i \bigg)^2 \bigg]} }\\ u(m) &= 0.00022161684162038723\\ \end{align}\]

Sy_mx_b2 = sum( (Y-m*X-b)*(Y-m*X-b) )
